import pandas as pd
from scipy.io import arff
from sklearn.preprocessing import LabelEncoder
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler

# Load ARFF files
train_file_path = "./archive/KDDTrain+.arff"
test_file_path = "./archive/KDDTest+.arff"

def load_arff(file_path):
    data, meta = arff.loadarff(file_path)
    df = pd.DataFrame(data)
    return df

# Load train and test datasets
train_df = load_arff(train_file_path)
test_df = load_arff(test_file_path)

# Display basic information about the datasets
print("Train dataset info:")
print(train_df.info())

print("\nTest dataset info:")
print(test_df.info())

# Encode categorical variables
le = LabelEncoder()
train_df['protocol_type'] = le.fit_transform(train_df['protocol_type'])
train_df['service'] = le.fit_transform(train_df['service'])
train_df['flag'] = le.fit_transform(train_df['flag'])
train_df['land'] = le.fit_transform(train_df['land'])
train_df['logged_in'] = le.fit_transform(train_df['logged_in'])
train_df['is_host_login'] = le.fit_transform(train_df['is_host_login'])
train_df['is_guest_login'] = le.fit_transform(train_df['is_guest_login'])
train_df['class'] = le.fit_transform(train_df['class'])

test_df['protocol_type'] = le.fit_transform(test_df['protocol_type'])
test_df['service'] = le.fit_transform(test_df['service'])
test_df['flag'] = le.fit_transform(test_df['flag'])
test_df['land'] = le.fit_transform(test_df['land'])
test_df['logged_in'] = le.fit_transform(test_df['logged_in'])
test_df['is_host_login'] = le.fit_transform(test_df['is_host_login'])
test_df['is_guest_login'] = le.fit_transform(test_df['is_guest_login'])
test_df['class'] = le.fit_transform(test_df['class'])

# Split features and target variable
X_train = train_df.drop('class', axis=1)
y_train = train_df['class']
X_test = test_df.drop('class', axis=1)

# Standardize the data

X_test = scaler.transform(X_test)
